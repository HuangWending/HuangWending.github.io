#include #include using namespace std; double f(double x) { return exp(x) - 2 * x; } double df(double x, double eps=1e-6) { double a = f(x), b = f(x + eps); return (b - a) / eps; } double integral(double a, double b, int n=1000) { double h = (b - a) / n; double sum = 0; for (int i = 0; i < n; i++) { double x = a + i * h; sum += f(x) * h; } return sum; } int main() { double x, a, b; cin >> x >> a >> b; cout << "f(" << x << ") = " << f(x) << endl; cout << "f'(" << x << ") = " << df(x) << endl; cout << "Integral[" << a << ", " << b << "] f(x) dx = " << integral(a, b) << endl; return 0; }

Java微积分(高等数学)计算程序，以下为代码

import java.util.Scanner; public class CalculusCalculator { public static double f(double x) { return Math.exp(x) - 2 * x; } public static double df(double x, double eps) { double a = f(x), b = f(x + eps); return (b - a) / eps; } public static double integral(double a, double b, int n) { double h = (b - a) / n; double sum = 0; for (int i = 0; i < n; i++) { double x = a + i * h; sum += f(x) * h; } return sum; } public static void main(String[] args) { Scanner scanner = new Scanner(System.in); double x = scanner.nextDouble(); double a = scanner.nextDouble(), b = scanner.nextDouble(); System.out.printf("f(%f) = %f%n", x, f(x)); System.out.printf("f'(%f) = %f%n", x, df(x, 1e-6)); System.out.printf("Integral[%f, %f] f(x) dx = %f%n", a, b, integral(a, b, 1000)); } }
