Java音乐播放器，有文件选择，播放列表、播放/暂停控制、音量控制

import java.io.File; import java.util.ArrayList; import javax.sound.sampled.AudioFormat; import javax.sound.sampled.AudioInputStream; import javax.sound.sampled.AudioSystem; import javax.sound.sampled.Clip; import javax.sound.sampled.DataLine; import javax.swing.JFileChooser; import javax.swing.JOptionPane; public class MusicPlayer { private ArrayList musicFiles; private Clip clip; private float volume; public MusicPlayer() { musicFiles = new ArrayList<>(); volume = 1.0f; } public void addMusicFile(String filePath) { musicFiles.add(filePath); } public void removeMusicFile(int index) { if (index >= 0 && index < musicFiles.size()) { musicFiles.remove(index); } } public void playMusic(int index) { if (index >= 0 && index < musicFiles.size()) { try { File audioFile = new File(musicFiles.get(index)); AudioInputStream audioStream = AudioSystem.getAudioInputStream(audioFile); AudioFormat format = audioStream.getFormat(); DataLine.Info info = new DataLine.Info(Clip.class, format); clip = (Clip) AudioSystem.getLine(info); clip.open(audioStream); clip.start(); setVolume(volume); } catch (Exception e) { JOptionPane.showMessageDialog(null, "Error playing music file: " + e.getMessage()); } } } public void stopMusic() { if (clip != null) { clip.stop(); } } public void pauseMusic() { if (clip != null) { clip.stop(); } } public void setVolume(float volume) { if (clip != null) { this.volume = volume; clip.setLevel(volume); } } public float getVolume() { return volume; } public static void main(String[] args) { MusicPlayer player = new MusicPlayer(); JFileChooser fileChooser = new JFileChooser(); fileChooser.setMultiSelectionEnabled(true); int result = fileChooser.showOpenDialog(null); if (result == JFileChooser.APPROVE_OPTION) { File[] files = fileChooser.getSelectedFiles(); for (File file : files) { player.addMusicFile(file.getAbsolutePath()); } } player.playMusic(0); } }

Java用API获取IP地址所在国家/地区，先安装Apache HttpClient库，以下为代码

org.apache.httpcomponents httpclient 4.5.10

import org.apache.http.HttpResponse; import org.apache.http.client.HttpClient; import org.apache.http.client.methods.HttpGet; import org.apache.http.impl.client.HttpClientBuilder; import org.apache.http.util.EntityUtils; import org.json.JSONObject; public class IPLocation { public static void main(String[] args) { try { HttpClient client = HttpClientBuilder.create().build(); HttpGet request = new HttpGet("http://ip.api.com/json/?lang=zh-cn"); HttpResponse response = client.execute(request); int statusCode = response.getStatusLine().getStatusCode(); if (statusCode == 200) { String responseBody = EntityUtils.toString(response.getEntity()); JSONObject json = new JSONObject(responseBody); String country = json.getString("country"); String region = json.getString("region"); System.out.println("所在国家/地区：" + country + "/" + region); } else { System.out.println("获取信息失败，状态码：" + statusCode); } } catch (Exception e) { e.printStackTrace(); } } }
